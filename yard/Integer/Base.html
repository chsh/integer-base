def parse(str, positional_chars)
  str = str.to_str.downcase

  sign = parse_sign! str
  abs = (
    case positional_chars.length
    when 1
      parse_unary_abs str, positional_chars.first
    else
      parse_positional_abs str, positional_chars
    end
  )
  
  (sign == :-) ? -abs : abs
end
